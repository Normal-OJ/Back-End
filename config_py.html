44 @field_validator('TESTING', mode='before') 

45 @classmethod 

46 def _lenient_testing(cls, v): 

47 '''Interpret ``1``, ``true``, and ``yes`` (case-insensitive) as True. 

48 

49 Any other value — including the common mistake of setting 

50 ``TESTING=false`` — is treated as False. 

51 ''' 

52 if isinstance(v, str): 

53 return v.lower() in ('1', 'true', 'yes') 

54 return v
